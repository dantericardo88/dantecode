  describe("Wave 2: Channel Version Tracking", () => {
    it("updateChannel increments version from 0 to 1", () => {
      expect(cp.getChannelVersion("tool-output")).toBe(0);
      cp.updateChannel("tool-output");
      expect(cp.getChannelVersion("tool-output")).toBe(1);
    });

    it("updateChannel increments version multiple times", () => {
      cp.updateChannel("model-output");
      cp.updateChannel("model-output");
      cp.updateChannel("model-output");
      expect(cp.getChannelVersion("model-output")).toBe(3);
    });

    it("tracks versions for multiple channels independently", () => {
      cp.updateChannel("tool-output");
      cp.updateChannel("tool-output");
      cp.updateChannel("model-output");
      expect(cp.getChannelVersion("tool-output")).toBe(2);
      expect(cp.getChannelVersion("model-output")).toBe(1);
    });

    it("getChannelVersion returns 0 for never-updated channel", () => {
      expect(cp.getChannelVersion("nonexistent")).toBe(0);
    });

    it("getAllChannelVersions returns empty object initially", () => {
      const versions = cp.getAllChannelVersions();
      expect(versions).toEqual({});
    });

    it("getAllChannelVersions returns all tracked channels", () => {
      cp.updateChannel("tool-output");
      cp.updateChannel("model-output");
      cp.updateChannel("model-output");
      const versions = cp.getAllChannelVersions();
      expect(versions).toEqual({
        "tool-output": 1,
        "model-output": 2,
      });
    });

    it("getAllChannelVersions returns a copy (mutations don't affect internal state)", () => {
      cp.updateChannel("test");
      const versions = cp.getAllChannelVersions();
      versions["test"] = 999;
      expect(cp.getChannelVersion("test")).toBe(1);
    });

    it("channel versions persist in checkpoint", async () => {
      cp.updateChannel("tool-output");
      cp.updateChannel("model-output");
      await cp.put({ step: 1 }, { source: "loop", step: 1 });

      const tuple = await cp.getTuple();
      expect(tuple!.checkpoint.channelVersions["tool-output"]).toBe(1);
      expect(tuple!.checkpoint.channelVersions["model-output"]).toBe(1);
    });

    it("channel versions restore on resume", async () => {
      cp.updateChannel("tool-output");
      cp.updateChannel("tool-output");
      await cp.put({ step: 1 }, { source: "loop", step: 1 });

      const resumed = new EventSourcedCheckpointer("/project", "sess-001", {
        writeFileFn: mockWrite,
        readFileFn: mockRead,
        mkdirFn: mockMkdir,
        readdirFn: mockReaddir,
        unlinkFn: mockUnlink,
      });

      await resumed.resume();
      expect(resumed.getChannelVersion("tool-output")).toBe(2);
    });

    it("channel versions continue incrementing after resume", async () => {
      cp.updateChannel("tool-output");
      await cp.put({ step: 1 }, { source: "loop", step: 1 });

      const resumed = new EventSourcedCheckpointer("/project", "sess-001", {
        writeFileFn: mockWrite,
        readFileFn: mockRead,
        mkdirFn: mockMkdir,
        readdirFn: mockReaddir,
        unlinkFn: mockUnlink,
      });

      await resumed.resume();
      resumed.updateChannel("tool-output");
      expect(resumed.getChannelVersion("tool-output")).toBe(2);
    });
  });
